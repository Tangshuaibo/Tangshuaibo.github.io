# 2. javascript的方法call()/apply()/bind()

参考文章：JavaScript 中 call()、apply()、bind() 的用法

以下例作为说明：

var A = {
    name: '小王',
    age: '15',
    show: function() {
        console.log('姓名：' + this.name + '；年龄：' + this.age)
    }
}

var B = {
    name: '老王',
    age: '45'
}

A.show.call(B)

运行结果：

姓名：老王；年龄：45

个人理解：A.show调用call()方法，修改了A对象的this的内存地址，使this指向B对象，因此扩大了B对象的作用域。

# 3. javascript的new命令

参考文章：js中的new做了什么？

以下例作为说明：

function Person () {
    this.name = name;
    this.age = age;
    this.sex = sex;
    
    this.sayName = function () {
        return this.name;
    };
}

var person = new Person("tom", 21, "male");

console.log(person.name);

对new命令执行过程进行分解：

var Obj = {};

Obj._proto_ =  Person.prototype;

Person.call(Obj);

# 4. javascript的继承

参考文章：详解ES5和ES6的继承

ES5的继承方式：

  • 原型链继承
  • 借用构造函数继承
  • 组合继承
  • 原型式继承
  • 寄生式继承
  • 寄生组合式继承（最佳方式）

参考文章：ES5的继承和ES6的继承有什么区别？

精简总结ES5与ES6继承的区别：

  • ES5的继承是先创建子类的实例对象this，然后再将父类的属性和方法添加到子类的this上（调用call()/new）；
  • ES6的继承是先创建父类的实例对象this（调用super()），然后再用子类的构造函数修改this。

# 5. 使用ES5的语法模拟面向对象语言的类与继承

例如，创建一个Animal类：

// 一个Animal类
var Animal = (function () {

    // 构造器
    function animal (name, color) {
        this.name = name;
        this.color = color;
    }

    // 原型方法
    animal.prototype = {
        setName: function (name) {
            this.name = name;
        },
        setColor: function (color) {
            this.color = color;
        },
        getName: function () {
            return this.name;
        },
        getColor: function () {
            return this.color;
        }
    };

    // 静态属性
    animal.movable = true;

    // 暴露构造器
    return animal;

})();

使用“寄生组合式继承”的方式（已经分解了new命令的执行过程），创建一个继承自Animal的Cat类：

// 一个Cat类，继承自Animal
var Cat = (function () {
    
    // 构造器
    function cat (name, color, sex) {
        Animal.call(this, name, color);
        this.sex = sex;
    }

    // 原型
    cat.prototype = {
        constructor: cat,
        __proto__: Animal.prototype,
        getSex: function () {
            return this.sex;
        }
    };

    // 静态属性
    cat.__proto__ = Animal;
    cat.flying = false;

    // 暴露构造器
    return cat;

})();

由于__proto__是浏览器的非标准属性，所以上例可以使用javascript的标准对象Object提供的方法修改：

参考文章：

  • Object.create()
  • Object.assign()
  • Object.setPrototypeOf()
// 一个Cat类，继承自Animal（使用Object的方法代替__proto__）
var Cat = (function () {
    
    // 构造器
    var cat = function (name, color, sex) {
        Animal.call(this, name, color);
        this.sex = sex;
    };
    
    // 原型
    cat.prototype = Object.create(Animal.prototype);
    Object.assign(cat.prototype, {
        constructor: cat,
        getSex: function () {
            return this.sex;
        }
    });
    
    // 静态属性
    Object.setPrototypeOf(cat, Animal);
    cat.flying = false;
    
    // 暴露构造器
    return cat;
    
})();

还可以模拟ES6继承的效果（先创建父类实例，再对其加工）：

// 一个Cat类，继承自Animal（先创建父类实例，再对其加工）
var Cat = (function () {
    
    // 构造器
    function cat (name, color, sex) {
        // 使用new命令创建父类实例并修改，如下：
        var animal = new Animal(name, color);
        animal.__proto__ = cat.prototype;
        // 或者拆解new命令的执行过程，如下：
        // var animal = {};
        // animal.__proto__ = cat.prototype;
        // Animal.call(animal, name, color);
        // 以上是对ES6的super()方法的执行效果的模拟
        animal.sex = sex;
        // return命令会改变实例对象的this的指向
        // 这里的this最终指向了animal
        return animal;
    }

    // 原型
    cat.prototype = {
        constructor: cat,
        __proto__: Animal.prototype,
        getSex: function () {
            return this.sex;
        }
    };

    // 静态属性
    cat.__proto__ = Animal;
    cat.flying = false;

    // 暴露构造器
    return cat;

})();

# 6. 使用ES5的语法模拟jQuery的trigger()方法

参考文章：创建和触发 events

下面以test.html与test.js的内容为例。

test.html的内容如下所示：

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <script src="./test.js"></script>
</head>
<body>
    <button id="btn-test">按钮1</button>
    <button id="btn-test-2">按钮2</button>
    <button id="btn-test-3">按钮3</button>
</body>
</html>

test.js的内容如下所示：

// 页面加载完成后执行
document.addEventListener('DOMContentLoaded', function () {

    /**
     *  使用Event构造函数创建自定义事件
     **/
    // 新建自定义事件
    var myEvent = new Event('test-event');
    // 添加对自定义事件的监听
    document.addEventListener('test-event', function (event) {
        console.log(event.type)
    }, false);
    // 添加对按钮（id为btn-test）点击事件的监听
    document.querySelector('#btn-test').addEventListener('click', function () {
        document.dispatchEvent(myEvent)
    })

    /**
     *  使用CustomEvent接口，可以为event对象添加自定义数据
     **/
    // 新建自定义事件
    var myEvent2 = new CustomEvent('test-event-2', {
        detail: {
            message: 'Destroy All Humans!'
        }
    });
    // 添加对自定义事件的监听
    document.addEventListener('test-event-2', function (event) {
        console.log(event.detail.message)
    }, false);
    // 添加对按钮（id为btn-test-2）点击事件的监听
    document.querySelector('#btn-test-2').addEventListener('click', function () {
        document.dispatchEvent(myEvent2)
    })

    /**
     *  触发浏览器内置事件
     **/
    // 下例模拟了内置的鼠标点击事件
    // 新建鼠标点击事件
    var myEvent3 = new MouseEvent('click', {
        'bubbles': true,
        'cancelable': true
    });
    // 添加对按钮（id为btn-test-3）点击事件的监听
    document.querySelector('#btn-test-3').addEventListener('click', function () {
        // 触发按钮（id为btn-test-2）的点击事件
        document.querySelector('#btn-test-2').dispatchEvent(myEvent3)
    });
})

# 7. javascript的getter/setter

参考文章：

  • getter
  • setter
  • Object.defineProperty()

在新对象初始化时定义getter/setter：

var obj1 = {
    _inner: '老王',
    get name() {
        return '名字 - '.concat(this._inner)
    },
    set name(value) {
        this._inner = value
    }
}

console.log(obj1.name) // output: 名字 - 老王

obj1.name = '隔壁老王'
console.log(obj1.name) // output: 名字 - 隔壁老王

// 使用delete操作符删除getter/setter
delete obj1.name
console.log(obj1.name) // output: undefined

使用defineProperty在现有对象上定义getter/setter：

var obj2 = {
    _inner: '老王'
}

Object.defineProperty(obj2, 'name', {
    configurable: true,
    enumerable: true,
    get: function () {
        return '名字 - '.concat(this._inner)
    },
    set: function (value) {
        this._inner = value
    }
})

console.log(obj2.name) // output: 名字 - 老王

obj2.name = '隔壁老王'
console.log(obj2.name) // output: 名字 - 隔壁老王

// 使用delete操作符删除getter/setter
delete obj2.name
console.log(obj2.name) // output: undefined

在Object.defineProperty()的属性描述符（descriptor）中，get/set不能和value/writable同时配置，否则会报错。

# 8. javascript的方法addEventListener()的参数

参考文章：EventTarget.addEventListener()

可用语法：

  • target.addEventListener(type, listener [, options]);
  • target.addEventListener(type, listener [, useCapture]);

options对象的可用参数：capture/once/passive。

useCapture是一个布尔值，而options是一个对象。

# 9. event对象的target与currentTarget属性的区别

参考文章：

  • Event
  • event.target 和 event.currentTarget 的区别

结论：event.target指向引起触发事件的元素，而event.currentTarget则是事件绑定的元素；只有触发事件的那个目标元素的event.target才等于event.currentTarget。

  1. 浏览器中事件传递的顺序是先捕获后冒泡；
  2. 假如目标元素能监听到click事件，那么事件传递过程中的相关元素都能监听click事件（除非阻止捕获或冒泡）；
  3. 方法stopImmediatePropagation()与stopPropagation()顾名思义“阻止推进”，可以阻止捕获或冒泡。
