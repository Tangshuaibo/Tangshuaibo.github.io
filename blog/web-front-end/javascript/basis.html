参考文章：

  • ES5的继承和ES6的继承有什么区别？

精简总结ES5与ES6继承的区别：

  • ES5的继承是先创建子类的实例对象this，然后再将父类的属性和方法添加到子类的this上（调用call()/new）；
  • ES6的继承是先创建父类的实例对象this（调用super()），然后再用子类的构造函数修改this。

# 5. 使用ES5的语法模拟面向对象语言的类与继承

例如，创建一个Animal类：

// 一个Animal类
var Animal = (function () {

    // 构造器
    function animal (name, color) {
        this.name = name;
        this.color = color;
    }

    // 原型方法
    animal.prototype = {
        constructor: animal,
        setName: function (name) {
            this.name = name;
        },
        setColor: function (color) {
            this.color = color;
        },
        getName: function () {
            return this.name;
        },
        getColor: function () {
            return this.color;
        }
    };

    // 静态属性
    animal.movable = true;

    // 暴露构造器
    return animal;

})();

使用“寄生组合式继承”的方式（已经分解了new命令的执行过程），创建一个继承自Animal的Cat类：

// 一个Cat类，继承自Animal
var Cat = (function () {
    
    // 构造器
    function cat (name, color, sex) {
        Animal.call(this, name, color);
        this.sex = sex;
    }

    // 原型
    cat.prototype = {
        constructor: cat,
        __proto__: Animal.prototype,
        getSex: function () {
            return this.sex;
        }
    };

    // 静态属性
    cat.__proto__ = Animal;
    cat.flying = false;

    // 暴露构造器
    return cat;

})();

由于__proto__是浏览器的非标准属性，所以上例可以使用javascript的标准对象Object提供的方法修改：

参考文章：

  • Object.create()
  • Object.assign()
  • Object.setPrototypeOf()
// 一个Cat类，继承自Animal（使用Object的方法代替__proto__）
var Cat = (function () {
    
    // 构造器
    var cat = function (name, color, sex) {
        Animal.call(this, name, color);
        this.sex = sex;
    };
    
    // 原型
    cat.prototype = Object.create(Animal.prototype);
    Object.assign(cat.prototype, {
        constructor: cat,
        getSex: function () {
            return this.sex;
        }
    });
    
    // 静态属性
    Object.setPrototypeOf(cat, Animal);
    cat.flying = false;
    
    // 暴露构造器
    return cat;
    
})();

# 6. 使用ES5的语法实现Promise

参考文章：

  • Promises/A+
  • 简单实现Promise
  • 从0到1实现Promise
  • 从零实现一个 Promise

主要代码如下：

var MyPromise = (function () {

    // 状态常量
    var PENDING = 'pending'
    var FULFILLED = 'fulfilled'
    var REJECTED = 'rejected'
    
    // 实例构造器
    var myPromise = function (executor) {
        this.state = PENDING
        this.data = undefined
        this.onFulFilledCallbacks = []
        this.onRejectedCallbacks = []
        var self = this
        var resolve = function (value) {
            if (self.state !== PENDING) return
            self.state = FULFILLED
            self.data = value
            self.onFulFilledCallbacks.forEach(function (callback) {
                callback()
            })
        }
        var reject = function (error) {
            if (self.state !== PENDING) return
            self.state = REJECTED
            self.data = error
            if (!self.onRejectedCallbacks.length) {
                throw new Error(self.data)
            }
            self.onRejectedCallbacks.forEach(function (callback) {
                callback()
            })
        }
        try {
            executor(resolve, reject)
        } catch (error) {
            reject(error)
        }
    }

    // 实例原型
    myPromise.prototype = {
        constructor: myPromise,
        then: function (onFulFilled, onRejected) {
            onFulFilled = typeof onFulFilled === 'function' ? onFulFilled : function (value) { return value }
            onRejected = typeof onRejected === 'function' ? onRejected : function (error) { throw error }
            var self = this
            var promise = new myPromise(function (resolve, reject) {
                switch (self.state) {
                case FULFILLED:
                    setTimeout(function () {
                        try {
                            var value = onFulFilled(self.data)
                            myPromise.resovePromise(promise, value, resolve, reject)
                        } catch (error) {
                            reject(error)
                        }
                    }, 0)
                    break
                case REJECTED:
                    setTimeout(function () {
                        try {
                            var error = onRejected(self.data)
                            myPromise.resovePromise(promise, error, resolve, reject)
                        } catch (error) {
                            reject(error)
                        }
                    }, 0)
                    break
                case PENDING:
                    self.onFulFilledCallbacks.push(function () {
                        setTimeout(function () {
                            try {
                                var value = onFulFilled(self.data)
                                myPromise.resovePromise(promise, value, resolve, reject)
                            } catch (error) {
                                reject(error)
                            }
                        }, 0)
                    })
                    self.onRejectedCallbacks.push(function () {
                        setTimeout(function () {
                            try {
                                var error = onRejected(self.data)
                                myPromise.resovePromise(promise, error, resolve, reject)
                            } catch (error) {
                                reject(error)
                            }
                        }, 0)
                    })
                    break
                default:
                    break
                }
            })
            return promise
        },
        catch: function (onRejected) {
            return this.then(undefined, onRejected)
        }
    }

    // 静态方法与属性
    myPromise.isThenable = function (obj) {
        return (typeof obj === 'object' || typeof obj === 'function') && typeof obj.then === 'function'
    }

    myPromise.resovePromise = function (promise, x, resolve, reject) {
        var called = false
        if (promise === x) return reject(new TypeError('promise circular reference'))
        if (!myPromise.isThenable(x)) return resolve(x)
        try {
            x.then(function (y) {
                if (called) return
                called = true
                myPromise.resovePromise(promise, y, resolve, reject)
            }, function (error) {
                if (called) return
                called = true
                reject(error)
            })
        } catch (error) {
            if (called) return
            called = true
            reject(error)
        }
    }

    // 暴露实例构造器
    return myPromise

})()

我们可以让上面的代码更面向对象一些：

var MyPromise = (function () {

    // 状态常量
    var PENDING = 'pending'
    var FULFILLED = 'fulfilled'
    var REJECTED = 'rejected'

    // 实例构造器
    function myPromise(executor) {
        this.state = PENDING
        this.data = undefined
        this.onFulFilledCallbacks = []
        this.onRejectedCallbacks = []
        this._start(executor)
    }

    // 实例原型
    myPromise.prototype = {
        constructor: myPromise,
        _start: function (executor) {
            try {
                executor(this.resolve.bind(this), this.reject.bind(this))
            } catch (error) {
                this.reject(error)
            }
        },
        resolve: function (value) {
            if(this.state !== PENDING) return
            this.state = FULFILLED
            this.data = value
            this.onFulFilledCallbacks.forEach(function (callback) {
                callback()
            })
        },
        reject: function (error) {
            if (this.state !== PENDING) return
            this.state = REJECTED
            this.data = error
            if (!this.onRejectedCallbacks.length) {
                throw new Error(this.data)
            }
            this.onRejectedCallbacks.forEach(function (callback) {
                callback()
            })
        },
        then: function (onFulfilled, onRejected) {
            onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : function (value) { return value }
            onRejected = typeof onRejected === 'function' ? onRejected : function (error) { throw error }
            var self = this
            var promise = new myPromise(function (resolve, reject) {
                var _getCallback = function (callback) {
                    return function () {
                        setTimeout(function () {
                            try {
                                var result = callback(self.data)
                                myPromise._resovePromise(promise, result)
                            } catch (error) {
                                reject(error)
                            }
                        }, 0)
                    }
                }
                var onFulFilledCallback = _getCallback(onFulfilled)
                var onRejectedCallback = _getCallback(onRejected)
                switch (self.state) {
                case PENDING:
                    self.onFulFilledCallbacks.push(onFulFilledCallback)
                    self.onRejectedCallbacks.push(onRejectedCallback)
                    break
                case FULFILLED:
                    onFulFilledCallback()
                    break
                case REJECTED:
                    onRejectedCallback()
                    break
                default:
                    break
                }
            })
            return promise
        },
        catch: function (onRejected) {
            return this.then(undefined, onRejected)
        }
    }

    // 静态属性与方法
    myPromise._isThenable = function (obj) {
        return (typeof obj === 'object' || typeof obj === 'function') && typeof obj.then === 'function'
    }
    myPromise._resovePromise = function (promise, x) {
        var called = false
        if (promise === x) return promise.reject(new TypeError('promise circular reference'))
        if (!myPromise._isThenable(x)) return promise.resolve(x)
        try {
            x.then(function (y) {
                if (called) return
                called = true
                myPromise._resovePromise(promise, y)
            }, function (error) {
                if (called) return
                called = true
                promise.reject(error)
            })
        } catch (error) {
            if (called) return
            called = true
            promise.reject(error)
        }
    }
    
    // 暴露实例构造器
    return myPromise

})()

# 7. 实现对JSONP的处理方法

参考文章：

  • JavaScript - 原生Jsonp及封装
  • 彻底弄懂jsonp原理及实现方法

这里以百度搜索建议API为例。

主要代码如下：

function jsonp (url, cbName, params) {
    // 禁止空url
    if (!url) return Promise.reject(new TypeError('url is necessary'))
    // 序列化参数
    let queries = ''
    for (const entry of Object.entries(params)) {
        queries = `${queries}&${entry[0]}=${entry[1]}`
    }
    const fullURL = `${url}?${queries}`
    // 建立脚本节点
    const scriptEle = document.createElement('script')
    scriptEle.setAttribute('type', 'text/javascript')
    scriptEle.setAttribute('src', fullURL)
    const bodyEle = document.body
    // 暴露Promise对象
    return new Promise(resolve => {
        window[cbName] = response => {
            bodyEle.removeChild(scriptEle)
            delete window[cbName]
            return resolve(response)
        }
        bodyEle.appendChild(scriptEle)
    })
}

# 9. javascript的getter/setter

参考文章：

  • getter
  • setter
  • Object.defineProperty()

在新对象初始化时定义getter/setter：

var obj1 = {
    _inner: '老王',
    get name() {
        return '名字 - '.concat(this._inner)
    },
    set name(value) {
        this._inner = value
    }
}

console.log(obj1.name) // output: 名字 - 老王

obj1.name = '隔壁老王'
console.log(obj1.name) // output: 名字 - 隔壁老王

// 使用delete操作符删除getter/setter
delete obj1.name
console.log(obj1.name) // output: undefined

使用defineProperty在现有对象上定义getter/setter：

var obj2 = {
    _inner: '老王'
}

Object.defineProperty(obj2, 'name', {
    configurable: true,
    enumerable: true,
    get: function () {
        return '名字 - '.concat(this._inner)
    },
    set: function (value) {
        this._inner = value
    }
})

console.log(obj2.name) // output: 名字 - 老王

obj2.name = '隔壁老王'
console.log(obj2.name) // output: 名字 - 隔壁老王

// 使用delete操作符删除getter/setter
delete obj2.name
console.log(obj2.name) // output: undefined

在Object.defineProperty()的属性描述符（descriptor）中，get/set不能和value/writable同时配置，否则会报错。

# 11. event对象的target与currentTarget属性的区别

参考文章：

  • Event
  • event.target 和 event.currentTarget 的区别
  • 组合模式
  • 责任链模式

结论：event.target指向引起触发事件的元素，而event.currentTarget则是事件绑定的元素；只有触发事件的那个目标元素的event.target才等于event.currentTarget。

  1. 浏览器中事件传递的顺序是先捕获后冒泡；
  2. 假如目标元素能监听到click事件，那么事件传递过程中的相关元素都能监听click事件（除非阻止捕获或冒泡）；
  3. 方法stopImmediatePropagation()与stopPropagation()顾名思义“阻止推进”，可以阻止捕获或冒泡。
